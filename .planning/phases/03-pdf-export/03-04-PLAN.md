```
#!/usr/bin/env bash
set -euo pipefail

# Smoke test for EXP-05: `npm install cora` must NOT download Chromium.
# We simulate npm install by running `npm pack` on the local package, then
# `npm install` the tarball into a fresh temp directory. The .npmrc shipped
# in the package must propagate `PLAYWRIGHT_SKIP_BROWSER_DOWNLOAD=1` so
# Playwright's postinstall skips the binary download.

PKG_DIR="$(cd "$(dirname "$0")/.." && pwd)"      # packages/cora
SMOKE_DIR="$(mktemp -d -t cora-smoke-XXXXXX)"
trap 'rm -rf "$SMOKE_DIR"' EXIT

echo "[smoke] PKG_DIR=$PKG_DIR"
echo "[smoke] SMOKE_DIR=$SMOKE_DIR"

# Build first — `npm pack` packs `dist/` per `files` array.
( cd "$PKG_DIR" && bun run build )

# Pack the local cora package into a tarball.
( cd "$PKG_DIR" && npm pack --pack-destination "$SMOKE_DIR" >/dev/null )
TARBALL="$(ls "$SMOKE_DIR"/cora-*.tgz | head -1)"
test -n "$TARBALL" || { echo "[smoke] FAIL: no tarball produced"; exit 1; }

# Isolate npm caches so we observe a clean install side-effect surface.
# Use the temp dir as HOME so any postinstall side effects land there.
HOME="$SMOKE_DIR" npm_config_cache="$SMOKE_DIR/npm-cache" \
  npm install --prefix "$SMOKE_DIR/install" "$TARBALL" \
  --ignore-scripts=false --no-audit --no-fund \
  >"$SMOKE_DIR/install.log" 2>&1 || {
    echo "[smoke] FAIL: npm install errored"; cat "$SMOKE_DIR/install.log"; exit 1;
  }

# Assertion 1: cora binary is reachable.
test -f "$SMOKE_DIR/install/node_modules/.bin/cora" \
  || { echo "[smoke] FAIL: cora bin not installed"; exit 1; }

# Assertion 2: NO Chromium directory under the isolated HOME.
# Playwright caches Chromium under $HOME/.cache/ms-playwright on Linux/mac
# and $LOCALAPPDATA/ms-playwright on Windows. Either presence is a failure.
test ! -d "$SMOKE_DIR/.cache/ms-playwright" \
  || { echo "[smoke] FAIL: Chromium was downloaded to $SMOKE_DIR/.cache/ms-playwright"; exit 1; }

# Assertion 3: TTF fonts are in the published tarball.
test -f "$SMOKE_DIR/install/node_modules/cora/dist/renderer/assets/fonts/NotoSans-Regular.ttf" \
  || { echo "[smoke] FAIL: TTF Noto Sans Regular missing from published package"; exit 1; }

# Assertion 4: .npmrc shipped in tarball (verifiable by listing tarball contents).
tar -tzf "$TARBALL" | grep -q '/.npmrc$' \
  || { echo "[smoke] FAIL: .npmrc not in published tarball — Playwright postinstall will fire"; exit 1; }

echo "[smoke] PASS: cora installs cleanly, no Chromium, fonts shipped, .npmrc shipped"
```
